**1. Status polling with useEffect:**
```typescript
useEffect(() => {
  if (!activeBatchId) return;

  const poll = async () => {
    try {
      const response = await apiClient.getUpscaleBatchDetail(activeBatchId);
      if (response.success && response.batch) {
        setBatch(response.batch);
        // Stop polling on terminal states
        if (['completed', 'failed', 'paused', 'cancelled'].includes(response.batch.status)) {
          clearInterval(intervalId);
        }
      }
    } catch (e) {
      // Silently handle polling errors
    }
  };

  poll(); // Initial fetch
  const intervalId = setInterval(poll, 4000); // 4 second interval

  return () => clearInterval(intervalId);
}, [activeBatchId]);
```
