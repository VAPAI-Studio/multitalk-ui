function StepIndicator({ steps, current, onStepClick }: { steps: BuilderStep[]; current: BuilderStep; onStepClick: (s: BuilderStep) => void }) {

  return (
    <div className="flex items-center gap-2 mb-8">
      {steps.map((s, i) => (
        <div key={s} className="flex items-center">
          <button
            onClick={() => i < currentIndex && onStepClick(s)}
            disabled={i > currentIndex}
            className={`px-3 py-1.5 rounded-xl text-sm font-medium transition-all ${
              s === current
                ? 'bg-slate-700 text-white shadow-md'
                : i < currentIndex
                ? 'bg-slate-200 text-slate-700 hover:bg-slate-300 cursor-pointer'
                : 'bg-gray-100 text-gray-400 cursor-not-allowed'

          </button>
          {i < steps.length - 1 && <div className="w-6 h-px bg-gray-300 mx-1" />}

      ))}
    </div>
  );
}

**Enrich from ComfyUI function** (async, optional, inside InspectStep):
```typescript
async function fetchObjectInfo(comfyUrl: string, classType: string): Promise<Record<string, unknown[]>> {
  try {
    const response = await fetch(`${comfyUrl}/object_info/${classType}`, {
      credentials: 'omit',
      signal: AbortSignal.timeout(5000),
    });
    if (!response.ok) return {};
    const data = await response.json();
    const nodeInfo = (data as Record<string, { input?: { required?: Record<string, unknown[]>, optional?: Record<string, unknown[]> } }>)[classType];
    if (!nodeInfo?.input) return {};
    return { ...(nodeInfo.input.required ?? {}), ...(nodeInfo.input.optional ?? {}) };
  } catch {
    return {};
  }
}
```
