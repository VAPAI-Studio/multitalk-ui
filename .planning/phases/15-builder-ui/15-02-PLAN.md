<feature>
  <name>Builder Utility Functions</name>
  <files>frontend/src/lib/builderUtils.ts, frontend/src/test/builderUtils.test.ts</files>
  <behavior>
    inferFieldType(value: unknown) → VariableInputType:
    - typeof value === 'boolean' → 'toggle'
    - Array.isArray(value) && all strings → 'dropdown'
    - typeof value === 'string' ending in .safetensors/.pth/.ckpt/.bin → 'text'
    - typeof value === 'string' ending in .png/.jpg/.jpeg/.webp → 'file-image'
    - typeof value === 'string' ending in .mp4/.mov/.webm → 'file-video'
    - typeof value === 'string' ending in .wav/.mp3/.flac → 'file-audio'
    - typeof value === 'string' (no extension match) → 'text'
    - typeof value === 'number' && Number.isInteger(value) → 'number'
    - typeof value === 'number' && !Number.isInteger(value) → 'slider'
    - fallback → 'text'

    extractClassTypes(nodes: ParsedNode[]) → string[]:
    - Returns unique class_type values from the nodes array
    - Skips duplicate class_types (uses Set)

    parseInstalledPackages(dockerfileContent: string) → Set<string>:
    - Matches lines containing 'custom_nodes/{name}' via regex /custom_nodes\/([^\s\\]+)/g
    - Returns Set of matched package names

    extractModelRefs(nodes: ParsedNode[]) → string[]:
    - For each node's configurable_inputs, checks input.name against MODEL_FIELDS set
    - Also checks input.value (string) extension against MODEL_EXTENSIONS set
    - Skips values starting with '{{' (already a placeholder)
    - Returns unique filenames

    checkModelPresence(refs: string[], manifest: ModelManifest) → ModelStatus[]:
    - For each ref, extracts basename (last path segment after '/')
    - Compares against manifest.models[].filename (basename, not full path)
    - Returns { filename: string, present: boolean }[]

    generateSlug(name: string) → string:
    - Mirrors backend generate_slug():  lowercase → strip special chars → replace spaces/hyphens with single '-' → trim leading/trailing '-'
    - 'My Feature Name!' → 'my-feature-name'
    - 'Multiple   Spaces' → 'multiple-spaces'
    - '  --leading-trailing--  ' → 'leading-trailing'

    derivePlaceholderKey(nodeId: string, inputName: string) → string:
    - '14' + 'ckpt_name' → '14_CKPT_NAME'
    - '3' + 'steps' → '3_STEPS'
    - replaces non-alphanumeric with '_' in the inputName uppercase portion
  </behavior>
